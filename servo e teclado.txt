#include <Servo.h>

Servo MOTOR; 
const int buttonPin = 8; 
bool buttonState = true; 
bool oldState = false;

void setup() {
  Serial.begin(9600);
  pinMode(buttonPin, INPUT_PULLUP); 
  pinMode(13, OUTPUT); 
  pinMode(12, OUTPUT); 
  MOTOR.attach(9);
  MOTOR.write(0);
}

void loop() {
    Serial.print(buttonState);
  
  if (digitalRead(buttonPin) && oldState == true) { 
    digitalWrite(12, LOW);      
    digitalWrite(13, HIGH);     
    MOTOR.write(90);
    oldState = false;
    delay(500);                 
  } 
  else if(digitalRead(buttonPin) && oldState == false){   
	oldState = true;
    digitalWrite(13, LOW);      
    digitalWrite(12, HIGH);     
    MOTOR.write(0);
    delay(500);  
  }
}
